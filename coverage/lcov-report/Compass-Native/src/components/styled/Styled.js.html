export const styleSwitch = (prop, cssIfValid, cssIfInvalid) => {
  switch (prop) {
    case null:
      return cssIfValid;
    case true:
      return cssIfValid;
    case false:
      return cssIfInvalid;
    default:
      return '';
  }
};
